function executeTrade(address counterparty, uint256 amount) external {
    // 1. 链上查询风险状态
    bool isSafe = riskRegistry.verifyAddress(
        counterparty,
        expectedRiskScore,
        "BLACK",
        merkleProof
    );

    // 2. 执行策略拦截
    require(isSafe, "对手方被合规策略拦截");

    // 3. 执行并生成不可篡改审计记录
    complianceEngine.recordDecision(msg.sender, counterparty, amount, 0);
    swap.execute(counterparty, amount);
}
